      total += (y===v('bah1Start'))?rate*((12-v('bah1Month')+1)/12):rate;
    }
    if(v('bah2Start')>0 && y>=v('bah2Start') && (v('bah2End')===0 || y<=v('bah2End')) && y<retireYr){
      const rate=v('bah2')*Math.pow(1+bah2Raise,Math.max(0,y-v('bah2Start')));
      total += (y===v('bah2Start'))?rate*((12-v('bah2Month')+1)/12):rate;
    }
    return total;
  }

  // ---- Debt Pmt ----
  function debtPaymentMonthly(y){
    let p=0;
    [['cc1'],['cc2'],['cc3'],['car1'],['car2'],['loan']].forEach(d=>{
      const id=d[0];
      const start=v(id+'Start'),end=v(id+'End'),pmtv=v(id+'Pmt');
      if(start>0 && y>=start && y<=end) p+=pmtv;
    });
    return p;
  }

  // ---- Simplified Expenses (Phase 1 & 2) + Private School + Tricare/Med + Inflation ----
  function expensesMonthly(year, forProjection){
    let e=0;
    if(year>=v('exp1Start') && year<=v('exp1End')){ e += v('exp1Bills') + v('exp1Disc'); }
    if(year>=v('exp2Start') && year<=v('exp2End')){ e += v('exp2Bills') + v('exp2Disc'); }
    
    // Private School Costs directly into expenses (Paid out of pocket)
    let privSchMo = 0;
    ['c1','c2','c3'].forEach(c => {
      let start = v(c+'privStart'), end = v(c+'privEnd');
      if(start > 0 && year >= start && year <= end) { 
          privSchMo += (v(c+'privTuition') + v(c+'privFees')) / 12; 
      }
    });
    e += privSchMo;

    // Add Healthcare Premiums
    if(v('tricareStart')>0 && year>=v('tricareStart') && (v('tricareEnd')===0 || year<=v('tricareEnd'))){ e += v('tricareMo'); }
    if(v('medInsStart')>0 && year>=v('medInsStart') && (v('medInsEnd')===0 || year<=v('medInsEnd'))){ e += v('medInsMo'); }
    if(v('denInsStart')>0 && year>=v('denInsStart') && (v('denInsEnd')===0 || year<=v('denInsEnd'))){ e += v('denInsMo'); }

    // Inflate
    if(forProjection){
      const inf = v('expInflation') / 100;
      const yrs = Math.max(0, year - curYear);
      e *= Math.pow(1 + inf, yrs);
    }
    return e;
  }
  
  function investMonthly(year){
    let c=0;
    if(year<=tspStopYear)c+=v('tspContribAD');
    if(v('rothMeStop')>0 && year<=v('rothMeStop'))c+=v('rothMe');
    if(v('rothWifeStop')>0 && year<=v('rothWifeStop'))c+=v('rothWife');
    if(v('c1stop')>0 && year<=v('c1stop'))c+=v('c1mo');
    if(v('c2stop')>0 && year<=v('c2stop'))c+=v('c2mo');
    if(v('c3stop')>0 && year<=v('c3stop'))c+=v('c3mo');
    if(v('gsTspStart')>0 && year>=v('gsTspStart')&&year<=v('gsTspEnd'))c+=v('gsTsp');
    c+=v('savMonthly');
    if(v('cryptoStart')>0 && year>=v('cryptoStart')&&year<=v('cryptoStop'))c+=v('cryptoMonthly');
    if(v('taxStart')>0 && year>=v('taxStart')&&year<=v('taxStop'))c+=v('taxMonthly');
    if(v('goldStart')>0 && year>=v('goldStart')&&year<=v('goldStop'))c+=v('goldMonthly');
    return c;
  }

  function adNet(y){
    let base=netPay0*Math.pow(1+payRaise,y-A);
    // Promo 1
    if(v('promo1Start')>0 && y>=v('promo1Start') && (v('promo1End')===0 || y<=v('promo1End'))){
        if(y===v('promo1Start')) base+=v('promo1Bump')*((12-v('promo1Month')+1)/12)*Math.pow(1+payRaise,y-v('promo1Start'));
        else base+=v('promo1Bump')*Math.pow(1+payRaise,y-v('promo1Start'));
    }
    // Promo 2
    if(v('promo2Start')>0 && y>=v('promo2Start') && (v('promo2End')===0 || y<=v('promo2End'))){
        if(y===v('promo2Start')) base+=v('promo2Bump')*((12-v('promo2Month')+1)/12)*Math.pow(1+payRaise,y-v('promo2Start'));
        else base+=v('promo2Bump')*Math.pow(1+payRaise,y-v('promo2Start'));
    }
    return base;
  }

  // ---- SBP and Pension Base ----
  const optSBP = txt('optSBP') === 'yes';
  const sbpAnnual = optSBP ? (retGross * 0.065) : 0;
  const retTaxable = retGross - sbpAnnual;
  const retNetBase = (retTaxable * (1 - fedTax)) / 12;

  // ============ ASSET PROJECTIONS (proj + actual + capgains + RMD) ============
  const rothRet=v('rothRet')/100,taxRet=v('taxRet')/100,savAPY=v('savAPY')/100,cryptoRet=v('cryptoRet')/100,goldRet=v('goldRet')/100;
  function contribFor(a,y){
    if(a==='Tsp')return (y<=tspStopYear)?v('tspContribAD'):((v('gsTspStart')>0&&y>=v('gsTspStart')&&y<=v('gsTspEnd'))?v('gsTsp'):0);
    if(a==='RothMe')return (v('rothMeStop')>0&&y<=v('rothMeStop'))?v('rothMe'):0;
    if(a==='RothWife')return (v('rothWifeStop')>0&&y<=v('rothWifeStop'))?v('rothWife'):0;
    if(a==='Tax')return (v('taxStart')>0&&y>=v('taxStart')&&y<=v('taxStop'))?v('taxMonthly'):0;
    if(a==='Sav')return v('savMonthly');
    if(a==='Crypto')return (v('cryptoStart')>0&&y>=v('cryptoStart')&&y<=v('cryptoStop'))?v('cryptoMonthly'):0;
    if(a==='Gold')return (v('goldStart')>0&&y>=v('goldStart')&&y<=v('goldStop'))?v('goldMonthly'):0;
    return 0;
  }
  function baseWithdrawFor(a,y){
    if(a==='Tsp')return (v('tspWdYear')>0&&y>=v('tspWdYear'))?v('tspWithdraw'):0;
    if(a==='RothMe')return (v('rothWdYear')>0&&y>=v('rothWdYear'))?v('rothWithdraw'):0;
    if(a==='RothWife')return (v('rothWifeWdYear')>0&&y>=v('rothWifeWdYear'))?v('rothWifeWithdraw'):0;
    if(a==='Tax')return (v('taxWdYear')>0&&y>=v('taxWdYear'))?v('taxWithdraw'):0;
    if(a==='Sav') {
        let w = 0;
        if(v('savWdYear')>0 && y>=v('savWdYear') && (v('savWdEnd')===0 || y<=v('savWdEnd'))) w += v('savWithdraw');
        if(v('savWdYear2')>0 && y>=v('savWdYear2') && (v('savWdEnd2')===0 || y<=v('savWdEnd2'))) w += v('savWithdraw2');
        return w;
    }
    if(a==='Crypto')return (v('cryptoWdYear')>0&&y>=v('cryptoWdYear'))?v('cryptoWithdraw'):0;
    if(a==='Gold')return (v('goldWdYear')>0&&y>=v('goldWdYear'))?v('goldWithdraw'):0;
    return 0;
  }

  const projRet={Tsp:tspRet,RothMe:rothRet,RothWife:rothRet,Tax:taxRet,Sav:savAPY,Crypto:cryptoRet,Gold:goldRet};
  const startBal={Tsp:v('tspBal'),RothMe:v('rothBalMe'),RothWife:v('rothBalWife'),Tax:v('taxBal'),Sav:v('savBal'),Crypto:v('cryptoBal'),Gold:v('goldBal')};
  const capGainsPct={Tax:v('taxCapGains')/100,Crypto:v('cryptoCapGains')/100,Gold:v('goldCapGains')/100};

  const proj={}, actual={}, actPct={}, capPaid={}, projWd={};
  let pBal={},aBal={}; ACCTS.forEach(a=>{pBal[a]=startBal[a];aBal[a]=startBal[a];});
  let pRolled=false,aRolled=false;
  
  for(let y=Y0;y<=YEND;y++){
    proj[y]={};actual[y]={};actPct[y]={};capPaid[y]={};projWd[y]={};
    ACCTS.forEach(a=>{
      let wd = baseWithdrawFor(a,y);
      
      // Auto-RMD check for TSP
      if(a === 'Tsp' && myAge(y) >= 73) {
         let div = RMD_DIVISORS[myAge(y)] || 6.0;
         let minWd = (pBal[a] / div) / 12;
         if(minWd > wd) wd = minWd;
      }
      projWd[y][a] = wd;

      const con=contribFor(a,y);
      // projected
      const mr=Math.pow(1+projRet[a],1/12)-1;
      let b=pBal[a];for(let m=0;m<12;m++){b=b*(1+mr)+con-wd;if(b<0)b=0;}pBal[a]=b;proj[y][a]=b;
      
      const balA=(useActuals)?actBal(a,y):null;
      const conA=(useActuals)?actCon(a,y):null;
      const conUsed=(conA!==null)?conA:con;
      if(balA!==null && y<=curYear){
        const netC=conUsed*12 - wd*12;
        actPct[y][a]=(aBal[a]>0)?(((balA-netC)/aBal[a])-1)*100:0;
        aBal[a]=balA;
      } else {
        let ab=aBal[a];for(let m=0;m<12;m++){ab=ab*(1+mr)+conUsed-wd;if(ab<0)ab=0;}aBal[a]=ab;actPct[y][a]=null;
      }
      actual[y][a]=aBal[a];
      if((a==='Tax'||a==='Crypto'||a==='Gold') && wd>0){capPaid[y][a]=wd*12*(capGainsPct[a]||0);} else {capPaid[y][a]=0;}
    });
    if(y===jbSell && tspRoll>0){if(!pRolled){pBal.Tsp-=pBal.Tsp*tspRoll;proj[y].Tsp=pBal.Tsp;pRolled=true;}if(!aRolled){aBal.Tsp-=aBal.Tsp*tspRoll;actual[y].Tsp=aBal.Tsp;aRolled=true;}}
  }

  // ---- Income / Cash Withdrawals Tracker ----
  function incomeParts(y){
    let mine=0,spouse=0,bits=[], sbpMo=0;
    if(y<retireYr){
      mine=adNet(y);bits.push('Active CW'+(y>=2030?'4':'3'));
    } else if(y===retireYr){
      const a=adNet(y);
      let activeMonths = Math.max(0, retireMonth - 1);
      let pensionMonths = 12 - activeMonths;
      mine=(a*activeMonths + retNetBase*pensionMonths)/12;
      bits.push('Active→Mil Retirement');
      if(optSBP) sbpMo = (sbpAnnual/12);
    } else {
      // Pension grows at pensionCOLA rate (not pay raise)
      mine = retNetBase * Math.pow(1+pensionCOLA, y-retireYr);
      mine += vaDis; bits.push('Pension+VA');
      if(optSBP) sbpMo = (sbpAnnual/12) * Math.pow(1+pensionCOLA, y-retireYr);
      if(y>=ssMeYear){mine+=ssMe;bits.push('MySS');}
      // Post-retirement extra work / part-time
      if(myWork>0){ mine+=myWork; bits.push('ExtraWork'); }

      // Withdrawals driven from actual logic if used, else projected
      const source = (useActuals && y <= curYear) ? actual : projWd;

      if(projWd[y] && projWd[y].Tsp > 0){ mine += projWd[y].Tsp * 0.85; bits.push('TSP-WD'); }
      if(projWd[y] && projWd[y].RothMe > 0){ mine += projWd[y].RothMe; bits.push('Roth-WD'); }
      if(projWd[y] && projWd[y].RothWife > 0){ spouse += projWd[y].RothWife; bits.push('SpRoth-WD'); }
    }

    // GS / Federal Civilian Job income (net after fed + state tax)
    if(gsGross>0 && gsIncomeStart>0 && y>=gsIncomeStart && y<=gsIncomeEnd){
      const gsNet = (gsGross/12) * (1 - gsFedTax - gsStateTax);
      mine += gsNet; bits.push('GS Job');
    }

    // Add savings, tax, crypto, gold withdrawals directly to income
    if(projWd[y] && projWd[y].Sav > 0){ mine += projWd[y].Sav; bits.push('Sav-WD'); }
    if(projWd[y] && projWd[y].Tax > 0){ mine += projWd[y].Tax * (1 - v('taxCapGains')/100); bits.push('Tax-WD'); }
    if(projWd[y] && projWd[y].Crypto > 0){ mine += projWd[y].Crypto * (1 - v('cryptoCapGains')/100); bits.push('Crypto-WD'); }
    if(projWd[y] && projWd[y].Gold > 0){ mine += projWd[y].Gold * (1 - v('goldCapGains')/100); bits.push('Gold-WD'); }

    if(y>=wifeStart && y<=wifeEnd){spouse+=wifeIncome;bits.push('SpouseWork');}
    if(y>=ssWifeYear){spouse+=ssWife;bits.push('SpouseSS');}
    return {mine, spouse, status:bits.join(', '), sbpMo};
  }

  // ---- Home equity / Debt ----
  function homeEquity(y){
    if(jbBuy>0 && y>=jbBuy && (jbSell===0 || y<=jbSell)){const frac=(y-jbBuy+1)/Math.max(1,(jbSell-jbBuy+1));return v('equity2033')*frac;}
    if(pcs2Buy>0 && y>=pcs2Buy && (pcs2Sell===0 || y<=pcs2Sell)){const frac=(y-pcs2Buy+1)/Math.max(1,(pcs2Sell-pcs2Buy+1));return v('pcs2EquityRoll')*frac;}
    if(y>=wakeBuy){const yrs=y-wakeBuy;const appr=townPrice*(Math.pow(1.03,yrs)-1);const prin=Math.min(townLoan,(townPI*0.30)*12*yrs);return totalDown+appr+prin;}
    return 0;
  }
  function amort(b0,pmtv,rate,fromY,toY){let b=b0;const r=rate/100/12;for(let yy=fromY;yy<=toY;yy++)for(let m=0;m<12;m++){b=b*(1+r)-pmtv;if(b<0)b=0;}return b;}
  function debtRemaining(y){let t=0;[['cc1'],['cc2'],['cc3'],['car1'],['car2'],['loan']].forEach(d=>{const id=d[0];t+=amort(v(id+'Bal'),v(id+'Pmt'),v(id+'Rate'),v(id+'Start'),y);});return t;}
  function mortgageRemaining(y){let t=0;
    if(jbBuy>0&&y>=jbBuy&&(jbSell===0||y<=jbSell)) t+=amort(jbLoan,jbPI,v('rate'),jbBuy,y);
    if(pcs2Buy>0&&y>=pcs2Buy&&(pcs2Sell===0||y<=pcs2Sell)) t+=amort(pcs2Loan,pcs2PI,v('pcs2Rate'),pcs2Buy,y);
    if(y>=wakeBuy) t+=amort(townLoan,townPI,townRate,wakeBuy,y);
    return t;
  }
  
  // ---- Accumulated Cash Flow Logic ----
  const accCashFlow = {};
  let runningCF = 0;
  for(let y=2026;y<=YEND;y++){
    if(y<2027) { accCashFlow[y] = 0; continue; }
    const ip=incomeParts(y);
    const spouse=showWife?ip.spouse:0;
    const combined=ip.mine+spouse;
    const bah=bahForYear(y);
    const home=homeForYear(y);
    const util=utilForYear(y);
    const debtPmt=debtPaymentMonthly(y);
    const exp=expensesMonthly(y, true); 
    const inv=investMonthly(y);
    const disc=combined+bah-home-util-debtPmt-exp-inv;
    runningCF += (disc*12);
    accCashFlow[y] = runningCF;
  }

  // ============ DASHBOARD ============
  function nw(y){
    const s= (useActuals && y <= curYear) ? (actual[y]||{}) : (proj[y]||{});
    const retire=(s.Tsp||0)+(s.RothMe||0)+(s.RothWife||0);
    const tax=s.Tax||0,sav=s.Sav||0,crypto=s.Crypto||0,gold=s.Gold||0;
    const eq=homeEquity(y);
    
    const consDebt=debtRemaining(y);
    const mortDebt=mortgageRemaining(y);
    const totalDebt=consDebt+mortDebt;
    
    const accCF = accCashFlow[y] || 0;
    const liquid = sav + tax + gold + accCF; // Cash flow affects liquidity directly
    
    const netNoMort = liquid + crypto + retire - consDebt;
    const netWithMort = netNoMort + eq - mortDebt;
    
    return {sav,tax,crypto,gold,retire,eq,consDebt,mortDebt,totalDebt,accCF,liquid,netNoMort,netWithMort};
  }
  function card(title,n,label){return `<div class="dashcard"><h4>${title}</h4>
    <div class="dashrow"><span>Liquid Cash (Sav+Tax+Gold+CF)</span><span style="color:#6fcf97;">${fmt(n.liquid)}</span></div>
    <div class="dashrow"><span>Crypto (volatile)</span><span>${fmt(n.crypto)}</span></div>
    <div class="dashrow"><span>Retirement (TSP+Roth)</span><span>${fmt(n.retire)}</span></div>
    <div class="dashrow"><span>Home Equity</span><span>${fmt(n.eq)}</span></div>
    <div class="dashrow"><span>Consumer Debt</span><span style="color:#ffb3b3;">-${fmt(n.consDebt)}</span></div>
    <div class="dashrow"><span>Mortgage Debt</span><span style="color:#ffb3b3;">-${fmt(n.mortDebt)}</span></div>
    <div class="dashtotal" style="font-size:14px; color:#cbd5e1; margin-bottom:4px;">Net Worth (Excl. Home): ${fmt(n.netNoMort)}</div>
    <div class="dashtotal">Projected Net Worth: ${fmt(n.netWithMort)}</div></div>`;}
  document.getElementById('dashboard').innerHTML=card('Current ('+curYear+')',nw(curYear),'Net Worth')+card('Age '+myAge(2040)+' ('+2040+')',nw(2040),'Projected Net Worth')+card('Age '+myAge(2060)+' ('+2060+')',nw(2060),'Estimated Net Worth');

  window._cf={incomeParts,expensesMonthly,investMonthly,homeForYear,homeExpenseForYear,homePIForYear,utilForYear,bahForYear,debtPaymentMonthly,myAge,wAge,locked,jbHomeMonthly,pcs2HomeMonthly,townHomeMonthly,selTownKey,TOWNS,jbBuy,jbSell,wakeBuy,retireYr,util1Loc:txt('util1Loc'),util2Loc:txt('util2Loc'),util3Loc:txt('util3Loc'),totalDown,townPI,townRate,wifeStart,wifeEnd,wifeIncome,retGross,payRaise,pensionCOLA,fedTax,vaDis,myWork,gsGross,gsFedTax,gsStateTax,gsIncomeStart,gsIncomeEnd,townExpenses,netTspForDown,optSBP,sbpAnnual};
  window._ctx={v,curYear,myAge,wAge,jbSell,proj,actual,actPct,capPaid,projWd,locked,projRet,useActuals,retireYr,accCashFlow};

  attachAutoSave();
  autoSave();
  buildSectionGroups();
}

  html+='</details></div>';

  html+=`<div class="panel"><details id="det_edu_combined" ${getOpen('det_edu_combined')}><summary><h2>5d. Combined Undergrad + Graduate Total Cost</h2></summary>`;
  html+='<div class="table-wrapper"><table><tr><th>Child</th><th>Undergrad Cost</th><th>Grad Cost</th><th>Private K-12 Cost</th><th>Total Education Cost</th><th>Total Out of Pocket</th></tr>';
  let gc=0,go=0;
  kids.forEach(k=>{
    const d=kidData[k.id];
    const gS=v(k.id+'gStart'), gE=v(k.id+'gEnd');
    const gT=v(k.id+'gTuition'), gRB=v(k.id+'gRB');
    const grbS=v(k.id+'grbStart'), grbE=v(k.id+'grbEnd');
    let gradCost=0, gradOOP=0;
    
    let simLeft=d.leftover||0, simSL=d.scholGrad;
    const simGY = Math.max(0,gE-gS+1);
    const simSP = simGY>0?simSL/simGY:0;

    if(gS>0 && gE>=gS){
      for(let y=gS;y<=gE;y++){
        let c=gT + ((y>=grbS && y<=grbE)?gRB:0);
        gradCost+=c;
        let sc=Math.min(simSP,c,simSL); simSL-=sc;
        let rem=c-sc;
        let u=Math.min(simLeft,rem); simLeft-=u;
        gradOOP+=(rem-u);
      }
    }
    
    let simUgOOP=0;
    let simPool=d.f529, simGM=gm, simSLUG=d.scholUG;
    const simSPUG=d.ugY>0?d.scholUG/d.ugY:0;
    if(d.ugY>0){
      for(let y=d.ugS;y<=d.ugE;y++){
        let c=d.ugT + ((y>=d.rbS && y<=d.rbE)?d.ugRB:0);
        let giU=0, gmU=Math.min(9,simGM);
        if(gmU>0){giU=Math.min(d.ugT,giTuition)+Math.min(((y>=d.rbS && y<=d.rbE)?d.ugRB:0),gmU*giMHA);simGM-=gmU;}
        let rem=Math.max(0,c-giU);
        let sc=Math.min(simSPUG,rem,simSLUG); simSLUG-=sc; rem-=sc;
        let u=Math.min(simPool,rem); simPool-=u; rem-=u;
        simUgOOP+=rem;
      }
    }
    
    let privCost = 0;
    let privS = v(k.id+'privStart'), privE = v(k.id+'privEnd');
    if(privS > 0 && privE >= privS) {
        privCost = (v(k.id+'privTuition')+v(k.id+'privFees')) * (privE - privS + 1);
    }

    const tot = d.ugCost + gradCost + privCost;
    const totOOP = simUgOOP + gradOOP + privCost; // Private is 100% OOP
    gc+=tot; go+=totOOP;
    html+=`<tr><td><b>${k.name}</b></td><td>${fmt(d.ugCost)}</td><td>${fmt(gradCost)}</td><td>${fmt(privCost)}</td><td><b>${fmt(tot)}</b></td><td class="${totOOP>0?'warn':'good'}">${fmt(totOOP)}</td></tr>`;
  });
  html+=`<tr style="background:#fff8e1"><td><b>ALL 3 CHILDREN</b></td><td>—</td><td>—</td><td>—</td><td><b>${fmt(gc)}</b></td><td class="${go>0?'warn':'good'}"><b>${fmt(go)}</b></td></tr>`;
  html+='</table></div></details></div>';

  document.getElementById('grp_edu').innerHTML=html;
}

// Wrapper for currency symbol
function applyCurrencyStyles() {
    document.querySelectorAll('input[type="number"]').forEach(el => {
        // Skip dates, ages, percents, terms
        const skip = ['age', 'year', 'month', 'pct', 'raise', 'ret', 'growth', 'inflation', 'rate', 'term', 'gibmonths', 'start', 'end'];
        if (skip.some(s => el.id.toLowerCase().includes(s))) return;
        
        // Prevent wrapping multiple times if loaded repeatedly
        if (el.parentNode.classList.contains('curr-wrap')) return;
        
        const w = document.createElement('span');
        w.className = 'curr-wrap';
        el.parentNode.insertBefore(w, el);
        w.appendChild(el);
    });
}


// ===== SIDEBAR NAVIGATION =====
function sideNav(type, id) {
  document.querySelectorAll('.sidebar-link').forEach(el => el.classList.remove('active'));
  if (type === 'input') {
    const el = document.getElementById('sl-' + id);
    if (el) el.classList.add('active');
    const det = document.getElementById('det_core');
    if (det) det.open = true;
    showTab(isNaN(id) ? id : parseInt(id));
    setTimeout(() => {
      const inp = document.getElementById('det_core');
      if (inp) inp.scrollIntoView({behavior:'smooth', block:'start'});
    }, 50);
  } else if (type === 'result') {
    const el = document.getElementById('sl-' + id);
    if (el) el.classList.add('active');
    showGroup(id);
    setTimeout(() => {
      const res = document.getElementById('results');
      if (res) res.scrollIntoView({behavior:'smooth', block:'start'});
    }, 50);
  } else if (type === 'dash') {
    const el = document.getElementById('sl-dash');
    if (el) el.classList.add('active');
    window.scrollTo({top: 0, behavior: 'smooth'});
  }
}
